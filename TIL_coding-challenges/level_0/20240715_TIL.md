    if angle < 90:

        answer = 2
    elif angle > 90 and angle < 180:
        answer = 3

# 배열의 평균값
def solution(numbers):
    sum = 0
    for number in numbers:
        sum += number
    answer = sum / len(numbers)
    return answer
```

# 짝수의 합
def solution(n):
    sum = 0
    while n > 0:
        if n % 2 == 0:
            sum += n
        n -= 1
    
    return sum
```
